127 /*
128 * LVGL Font Converter Format Specification:
129 * According the specification there are 4 formats:
130 * https://github.com/lvgl/lv_font_conv/blob/master/doc/font_spec.md
131 *
132 * For simplicity introduce "relative code point":
133 * rcp = codepoint - range_start
134 *
135 * and a search function:
136 * search a "value" in an "array" and returns the index of "value".
137 *
138 * Format 0 tiny:
139 * unicode_list == NULL && glyph_id_ofs_list == NULL
140 * glyph_id = glyph_id_start + rcp
141 *
142 * Format 0 full:
143 * unicode_list == NULL && glyph_id_ofs_list != NULL
144 * glyph_id = glyph_id_start + glyph_id_ofs_list[rcp]
145 *
146 * Sparse tiny:
147 * unicode_list != NULL && glyph_id_ofs_list == NULL
148 * glyph_id = glyph_id_start + search(unicode_list, rcp)
149 *
150 * Sparse full:
151 * unicode_list != NULL && glyph_id_ofs_list != NULL
152 * glyph_id = glyph_id_start + glyph_id_ofs_list[search(unicode_list, rcp)]
153 */
154
156 const uint16_t * unicode_list;
157
162 const void * glyph_id_ofs_list;
163
165 uint16_t list_length;
166
168 lv_font_fmt_txt_cmap_type_t type;
169} lv_font_fmt_txt_cmap_t;

187typedef struct {

189 1. Get the `glyph_id_left` and `glyph_id_right` from `lv_font_fmt_txt_cmap_t
190 2. Get the class of the left and right glyphs as `left_class` and `right_class`
191 left_class = left_class_mapping[glyph_id_left];
192 right_class = right_class_mapping[glyph_id_right];
193 3. value = class_pair_values[(left_class-1)*right_class_cnt + (right_class-1)]
